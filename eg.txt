class Bootstrap extends Zend_Application_Bootstrap_Bootstrap
{
    protected function _initControllers()
    {
        $this->bootstrap('FrontController');
        $this->frontController->setControllerDirectory(APPLICATION_PATH . '/controllers', 'default');
        $this->frontController->addModuleDirectory(APPLICATION_PATH . '/modules');
        $this->frontController->throwExceptions(false);  
        Zend_Registry::set('baseUrl',$this->frontController->getBaseUrl());
        Zend_Session::start();
        /*$router = $this -> _front -> getRouter();
        $config = new Zend_Config_Ini($this->_root . '/application/config/route.ini','route');
        $router -> addConfig($config, 'routes');*/
    }
   
    protected function _initLogger()
    {
        $config = new Zend_Config_Ini(APPLICATION_PATH."/configs/application.ini","logger");
        if($config->type == 'stream'){
            $writer = new Zend_Log_Writer_Stream($config->file);
            $logger = new Zend_Log($writer);
        }else{
            $writer = new Zend_Log_Writer_Firebug();
            $logger = new Zend_Log($writer);
        }
        $filter = new Zend_Log_Filter_Priority((int)$config->priority);
        $logger->addFilter($filter);
        Zend_Registry::set('logger',$logger);
    }
   
    protected function _initCache()
    {
        $config = new Zend_Config_Ini(APPLICATION_PATH."/configs/application.ini","cache");
        $frontendOptions = array('lifetime' => $config -> lifetime,
                                 'automatic_serialization' => $config -> automatic_serialization);
        $backendOptions = array('cache_dir' => $config -> dir);
        $cache = Zend_Cache::factory('Core', 'File', $frontendOptions, $backendOptions);
        Zend_Registry::set('cache',$cache);
    }
   
    protected function _initSession()
    {
        $config = new Zend_Config_Ini(APPLICATION_PATH."/configs/application.ini","auth");
        Zend_Registry::set('keys',$config->keys);
    }
   
    protected function _initDB()
    {
        $resources = $this->getPluginResource('db');
        $db = $resources->getDbAdapter();
        Zend_Registry::set('db',$db);
    }
