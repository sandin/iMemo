  4  * 
  5  * @param commandType
  6  * 
  7  * @return 
  8  */
  9 function CommandFactory(commandType)
 10 {
 11   var command = ucwords(commandType);
 12   command += 'Command';
 13 
 14   //确认所请求的Command类是存在的
 15   if (eval('typeof ' + command + ' != "undefined"')) {
 16 	return eval('new ' + command + '()');
 17   } else {
 18 	//如果不存在,则返回一个超类做替代,超类不调用任何回调函数
 19 	return new Command();
 20   }
 21 }
