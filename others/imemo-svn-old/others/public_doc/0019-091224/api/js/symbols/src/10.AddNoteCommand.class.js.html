  9 function AddNoteCommand(params)
 10 {
 11   Command.call(this);
 12   this.name = 'addNote';
 13 }
 14 AddNoteCommand.prototype = new Command();
 15 
 16 /** 
 17  * Command addNote,Status success 回调函数
 18  * 
 19  * @return 
 20  */
 21 AddNoteCommand.prototype.success = function()
 22 {
 23   if (this.params.statusText && this.params.statusText.toLowerCase() == 'success') {
 24 	try 
 25 	{
 26 	  //console.log(this.params.ajax.responseText,'responseText');
 27 	  var noteObject = JSON.parse(this.params.responseText); 
 28 	  var note = new Note();
 29 	  note.makeNoteHTML(noteObject);
 30 	}
 31 	catch(err) 
 32 	{
 33 	  txt =	 'Error: Response data is not a JSON!\n\n';
 34 	  txt += 'Code: 0x0000;\n\n';
 35 	  txt += 'Error name: ' + err.name + '\n\n';
 36 	  txt += 'Error message: ' + err.message + '\n\n';
 37 	  txt += 'Description: ' + err.description + '\n\n';
 38 	  
 39 	  if (console && console != null) {
 40 		console.log(txt);
 41 	  } else {
 42 		alert(txt);
 43 	  }
 44 	}
 45   }
 46 }

 53 AddNoteCommand.prototype.beforeSubmit = function()
 54 {
 55   var note_content = $('.ajax-add-note',this.params.jqForm).attr('value');
 56 
 57   //如果用户没有输入任何内容,则取消Ajax请求
 58   if (note_content == 'undefined' || note_content == '')
 59   {
 60 	return false;
 61   }
 62 }
