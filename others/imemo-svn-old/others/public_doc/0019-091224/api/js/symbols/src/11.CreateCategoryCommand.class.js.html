 16 /** 
 17  * Status success 回调函数
 18  * 
 19  * @return 
 20  */
 21 CreateCategoryCommand.prototype.success = function()
 22 {
 23   if (this.params.statusText && this.params.statusText.toLowerCase() == 'success') {
 24 	try 
 25 	{
 26 	  //console.log(this.params.ajax.responseText,'responseText');
 27 	  var noteObject = JSON.parse(this.params.responseText); 
 28 	  if (noteObject.data) {
 29 		//reload tab zoom
 30 		$('#settings-tabs').tabsExtra({'reload':true});
 31 	  }
 32 	}
 33 	catch(err) 
 34 	{
 35 	  txt =	 'Error: Response data is not a JSON!\n\n';
 36 	  txt += 'Code: 0x0000;\n\n';
 37 	  txt += 'Error name: ' + err.name + '\n\n';
 38 	  txt += 'Error message: ' + err.message + '\n\n';
 39 	  txt += 'Description: ' + err.description + '\n\n';
 40 	  
 41 	  if (console && console != null) {
 42 		console.log(txt);
 43 	  } else {
 44 		alert(txt);
 45 	  }
 46 	}
 47   }
 48 }
