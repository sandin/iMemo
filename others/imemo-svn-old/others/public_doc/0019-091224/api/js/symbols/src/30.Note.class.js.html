 15 /** 
 16  * 根据note模板新建一个note的HTML,并更新到视图
 17  * 
 18  * @param Object oData
 19  * 
 20  * @return void
 21  */
 22 Note.prototype.makeNoteHTML = function (oData)
 23 {
 24   var $new_note = $('#js_note_template>li').clone(true);
 25   $new_note.removeAttr('id').removeAttr('style');
 26   var id = oData.data.category_id + ":" + oData.data.note_id;
 27   $new_note.attr('id',id);
 28   //console.log($new_note);
 29 
 30   $new_note.find('td').not('.n.s').html('');
 31   $new_note.find('.n_del>form>.note_id').attr('value',oData.data.note_id);
 32   //console.log( $new_note.find('.n_del>form>.note_id') );
 33   
 34   $new_note.find('.n_content').html(oData.data.content);
 35   $('.notes_list:visible').prepend($new_note);
 36 }
