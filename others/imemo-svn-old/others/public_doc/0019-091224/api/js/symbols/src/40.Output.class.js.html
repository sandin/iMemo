 12 /** 
 13  * 新建input的HTML,如果指定了target则直接append其中.
 14  * 
 15  * @param name
 16  * @param value
 17  * @param $target
 18  * 
 19  * @return

 21 Output.prototype.makeInputHTML = function(name, value, $target)
 22 {
 23   var $new_input = $('<input type="hidden"></input>');
 24   $new_input.attr('name',name);
 25   $new_input.attr('value',value);
 26   if ($target && $target.length != 0) {
 27 	$target.append($new_input);
 28   }
