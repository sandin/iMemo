  2 /** 
  3  * class BindDelButton implements IObserver
  4  * 
  5  * @return 
  6  */
  7 function BindDelButton()
  8 { 
  9   IObserver.call(this);
 10   this._delAllNote;
 11 }
 12 
 13 BindDelButton.prototype.update = function(){
 14   var sThis = this;
 15 
 16   $('.c_del').each(function(){
 17 	if (!$(this).hasClass('binded'))
 18 	{
 19 	  $(this).addClass('binded');
 20 	  var $tr = $(this).parent();
 21 	  var $a = $(this).children();
 22 	  $a.click(function(){
 23 	  // dialog
 24 	  $('<p>Delete notes of this category?</p>').dialog({
 25 			  title: "Comfirm",
 26 			  bgiframe: true,
 27 			  resizable: false,
 28 			  height:140,
 29 			  modal: true,
 30 			  overlay: {
 31 				  backgroundColor: '#000',
 32 				  opacity: 0.5
 33 			  },
 34 			  buttons: {
 35 				  'Just delete this category': function() {
 36 					sThis._delAllNote = 0
 37 					sThis.ajaxSend($tr,$a);
 38 					$(this).dialog('close');
 39 				  },
 40 				  'Delete notes': function() {
 41 					sThis._delAllNote = 1;
 42 					sThis.ajaxSend($tr,$a);
 43 					$(this).dialog('close');
 44 				  }
 45 			  }
 46 		});
 47 
 48 	  });//end click
 49 
 50 	}//fi
 51   });//end each
 52 }//end function
 53 
 54 BindDelButton.prototype.ajaxSend = function($tr,$a)
 55 {
 56   var id  = $tr.find('.c_id').text();
 57   var url = $a.attr('href');
 58   var data = {categorys_id : id,delAllNote:this._delAllNote};
 59   $.ajax({
 60 	type		: "POST",
 61 	url			: url,
 62 	data		: data,
 63 	beforeSend	: function (XMLHttpRequest) {
 64 	},
 65 	success		: function(msg){
 66 	  //reload tab zoom
 67 	  $('#settings-tabs').tabsExtra({'reload':true});
 68 	}
 69   });//end ajax
 70 }
