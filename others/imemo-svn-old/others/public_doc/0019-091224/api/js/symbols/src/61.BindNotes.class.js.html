 12 BindNotes.prototype.update = function(){
 13   var sThis = this;
 14 
 15   $('.notes_list').each(function(){
 16 	if (!$(this).hasClass('binded'))
 17 	{
 18 	  $(this).addClass('binded');
 19 
 20       // notes sort
 21       $(this).sortable({
 22                 handle   :   '.n_lable',
 23                 cursorAt : { cursor: 'crosshair',
 24                              top: -3, left: -3 },
 25                 update   : function(event,ui) {
 26                    helper = new LdsHelper();
 27 
 28                    //所有note顺序的id
 29                    all_arr        = $(this).sortable('toArray');
 30                    //当前移动的note id解析
 31                    curr_arr       = $(ui.item).attr('id').split(':');
 32                    //当前移动note_id
 33                    curr_note_id   = curr_arr[1];
 34                    curr_cate_id   = curr_arr[0];
 35                    //将category和note的混合id数组解析出只包含note_id的数组
 36                    all_notes_arr  = helper.parseIdArray(all_arr,'note_id');
 37                    //当前移动的note在顺序中的数组索引
 38                    curr_index     = arrayIndexOf(all_notes_arr,curr_note_id);
 39                    //移动后,前一note的id
 40                    front_note_id  = all_notes_arr[curr_index - 1];
 41                    if ((typeof front_note_id) == 'undefined') {
 42                         front_note_id = null;
 43                    }
 44 
 45                    //准备需要发送的数据
 46                    var data = {
 47                             index    : curr_note_id,
 48                             front    : front_note_id,
 49                             categorys: curr_cate_id
 50                    };
 51                    sThis.sendSortData(data);
 52                 }
 53       }); //.disableSelection();
 54 
 55 
 56 	}//fi
 57   });//end each
 58 }//end function
 59 
 60 BindNotes.prototype.sendSortData = function(data)
 61 {
 62   var url =  $('#sort_note_url').attr('href');
 63   $.ajax({
 64 	type		: "POST",
 65 	url			: url,
 66 	data		: data,
 67 	beforeSend	: function (XMLHttpRequest) {
 68 	},
 69 	success		: function(msg){
 70                    //console.log(msg);
 71 	}
 72   });//end ajax
 73 }
